function verificarRespuesta(respuesta, correcta = null) { let correcto = correcta !== null ? respuesta === correcta : respuesta === 5; const activo = document.querySelector('.seccion.activo'); const resultadoDiv = activo.querySelector('div[id^="resultado"]'); resultadoDiv.innerHTML = ''; const video = document.createElement('video'); video.src = correcto ? "videos/correcto.mp4" : "videos/incorrecto2.mp4"; video.autoplay = true; video.controls = true; // Para poder probar sonido video.width = 616; video.height = 480; video.muted = false; video.playsInline = true; video.volume = 1.0; resultadoDiv.appendChild(video); }
